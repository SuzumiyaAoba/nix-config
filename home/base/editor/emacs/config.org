*** Emacs Daemon

Emacs デーモンを使っていると GUI/CUI で期待通りの設定が行えないことがある。
Emacs デーモンを起動したとき =window-system= が =nil= になってしまっているため、
=hook= を使わずに設定を記述すると GUI 用の設定が実行されない。

そのため、こちらの[[https://minosjp.hatenablog.com/entry/2020/12/03/221536][記事]]を参考に、

- =scroll-bar-mode=
- =tool-bar-mode=
- =menu-bar-mode=

を無効化する設定を GUI で起動したときに走るように設定している。
それでもターミナルで Emacs を起動するとメニューバーが表示されてしまうことがある。

#+begin_src emacs-lisp
;; see: https://minosjp.hatenablog.com/entry/2020/12/03/221536
(defvar my-gui-setup-hook nil
  "List of functions to call at initializing Emacs as GUI.")

(defun my-func-called-at-gui-initialization (&optional frame)
  "Function called at GUI initialization."
  (unless frame (setq frame (selected-frame)))
  (with-selected-frame frame
    (when (display-graphic-p)
      (run-hooks 'my-gui-setup-hook)
      (modify-frame-parameters frame default-frame-alist)
      (remove-hook 'after-make-frame-functions #'my-func-called-at-gui-initialization))))

(defun my-gui-setup ()
  "Setup if Emacs is running on GUI."
  (add-hook 'my-gui-setup-hook
            (lambda()
              (scroll-bar-mode -1)
              (tool-bar-mode -1)
              (menu-bar-mode -1)))

  (my-func-called-at-gui-initialization))

(add-hook 'after-init-hook #'my-gui-setup)
#+end_src

(use-package emacs
  :init
  ;; スクロールバー非表示
  (scroll-bar-mode -1)
  ;; ツールバー非表示
  (tool-bar-mode -1)
  ;; メニューバー非表示
  (menu-bar-mode -1)

  ;; ビープ音を消す
  (setq ring-bell-function 'ignore)

  ;; プログラミング/テキストモードの場合に行番号表示
  (add-hook 'prog-mode-hook #'display-line-numbers-mode)
  (add-hook 'text-mode-hook #'display-line-numbers-mode)

  ;; ネイティブコンパイル時に発生したエラー・警告を無視
  (setq native-comp-async-report-warnings-errors nil)

  ;; カスタムファイルの出力先を設定
  (setq custom-file "~/.emacs.d/custom.el")
  (if (file-exists-p "~/.emacs.d/custom.el")
      (load custom-file))

  ;; スクロールを滑らかにする
  (setq scroll-margin 0
        scroll-conservatively 100000
        scroll-preserve-screen-position t)
  (setq scroll-preserve-screen-position t)
  (pixel-scroll-precision-mode t)

  ;; 言語、文字コードを設定
  (set-language-environment "Japanese")
  (prefer-coding-system 'utf-8)
  (set-default 'buffer-file-coding-system 'utf-8)

  ;; ファイル変更をバッファに自動反映する
  (setq global-auto-revert-non-file-buffers t)
  (global-auto-revert-mode)

  ;; ミニバッファの履歴を保存
  (savehist-mode)
  ;; 最近使ったファイルを覚えていく
  (recentf-mode)

  ;; 現在行をハイライト
  (global-hl-line-mode)

  ;; インデントをスペースにする
  (setq-default indent-tabs-mode nil)

  (setq make-backup-files nil
        auto-save-default nil
        delete-by-moving-to-trash t
        vc-follow-symlinks t)

  ;; ミニバッファの yes/no を y/n に変更
  (defalias 'yes-or-no-p 'y-or-n-p)

  (setq completion-cycle-threshold 3)
  (setq tab-always-indent 'complete)

  (xterm-mouse-mode t)
  (mouse-wheel-mode t)
  (global-set-key [mouse-4] '(lambda () (interactive) (scroll-down 3)))
  (global-set-key [mouse-5] '(lambda () (interactive) (scroll-up 3)))

  ;; clipboard
  (defun copy-from-osx ()
    (shell-command-to-string "pbpaste"))

  (defun paste-to-osx (text &optional push)
    (let ((process-connection-type nil))
      (let ((proc (start-process "pbcopy" "*Messages*" "pbcopy")))
        (process-send-string proc text)
        (process-send-eof proc))))

  (when (eq system-type 'darwin)
    (setq interprogram-cut-function 'paste-to-osx)
    (setq interprogram-paste-function 'copy-from-osx)))

  (defun my/highlight-indent-guides--bitmap-line (width height crep zrep)
